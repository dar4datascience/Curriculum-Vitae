  {id: 12, name: "Kubernetes", category: "DevOps", value: 70, icon: "☸️"}
]

bubbleWidth = 900
bubbleHeight = 600
bubbleCenterX = bubbleWidth * 0.5
bubbleCenterY = bubbleHeight * 0.5

bubbleNodes = bubbleData.map(d => ({
  ...d,
  x: bubbleCenterX + (Math.random() - 0.5) * bubbleWidth * 0.8,
  y: bubbleCenterY + (Math.random() - 0.5) * bubbleHeight * 0.8,
  r: 0,
  radius: Math.sqrt(d.value) * 6,
  fx: null,
  fy: null
}))

bubbleSimulation = d3.forceSimulation(bubbleNodes)
  .force('charge', d3.forceManyBody().strength(5))
  .force('collide', d3.forceCollide(d => d.r + 3))
  .force('x', d3.forceX(bubbleCenterX).strength(0.05))
  .force('y', d3.forceY(bubbleCenterY).strength(0.05))

bubbleChart = {
  const svg = d3.create("svg")
    .attr("width", bubbleWidth)
    .attr("height", bubbleHeight)
    .style("background", "linear-gradient(135deg, #667eea 0%, #764ba2 100%)")
    .style("border-radius", "15px");

  let focusedNode = null;

  const node = svg.selectAll(".node")
    .data(bubbleNodes)
    .join("g")
    .attr("class", "node")
    .style("cursor", "pointer");

  node.append("circle")
    .attr("r", 0)
    .style("fill", "#fff")
    .style("stroke", "#333")
    .style("stroke-width", 2)
    .transition()
    .duration(2000)
    .delay((d, i) => i * 100)
    .ease(d3.easeElasticOut)
    .attr("r", d => d.radius);

  node.append("text")
    .attr("text-anchor", "middle")
    .attr("dy", "0.35em")
    .style("font-size", d => `${d.radius * 0.6}px`)
    .style("pointer-events", "none")
    .text(d => d.icon);

  node.on("click", function(event, d) {
    event.stopPropagation();
    if (d === focusedNode) return;
    
    if (focusedNode) {
      d3.transition().duration(1500)
        .tween("reset", () => {
          const ir = d3.interpolateNumber(focusedNode.r, focusedNode.radius);
          return t => { focusedNode.r = ir(t); };
        });
    }
    
    focusedNode = d;
    node.style("opacity", n => n === d ? 1 : 0.3);
    
    d3.transition().duration(1500)
      .tween("zoom", () => {
        const ix = d3.interpolateNumber(d.x, bubbleCenterX);
        const iy = d3.interpolateNumber(d.y, bubbleCenterY);
        const ir = d3.interpolateNumber(d.r, 100);
        return t => {
          d.x = ix(t);
          d.y = iy(t);
          d.r = ir(t);
        };
      });
  });

  svg.on("click", () => {
    if (focusedNode) {
      d3.transition().duration(1500)
        .tween("reset", () => {
          const ir = d3.interpolateNumber(focusedNode.r, focusedNode.radius);
          return t => { focusedNode.r = ir(t); };
        });
      focusedNode = null;
      node.style("opacity", 1);
    }
  });

  bubbleSimulation.on("tick", () => {
    node.attr("transform", d => `translate(${d.x},${d.y})`);
    node.select("circle").attr("r", d => d.r || d.radius);
  });

  bubbleNodes.forEach((d, i) => {
    setTimeout(() => { d.r = d.radius; }, i * 100);
  });

  return svg.node();
}

Section 5: Stats & Metrics

statsChart = {
  const data = [
    { label: "Projects", value: 50, icon: "📁" },
    { label: "Commits", value: 5000, icon: "💻" },
    { label: "Stars", value: 250, icon: "⭐" },
    { label: "Years Exp", value: 6, icon: "📅" }
  ];
  
  const container = d3.create("div")
    .attr("class", "stats-grid");
  
  const cards = container.selectAll(".stat-card")
    .data(data)
    .join("div")
    .attr("class", "stat-card")
    .style("opacity", 0)
    .html(d => `
      <div style="font-size: 3em;">${d.icon}</div>
      <div class="stat-number">0</div>
      <div class="stat-label">${d.label}</div>
    `);
  
  // Animate in
  cards.transition()
    .duration(800)
    .delay((d, i) => i * 200)
    .style("opacity", 1);
  
  // Count up animation
  cards.each(function(d, i) {
    const card = d3.select(this);
    const numberEl = card.select(".stat-number");
    
    setTimeout(() => {
      d3.transition()
        .duration(2000)
        .tween("count", () => {
          const interpolate = d3.interpolateNumber(0, d.value);
          return t => {
            numberEl.text(Math.round(interpolate(t)));
          };
        });
    }, i * 200);
  });
  
  return container.node();
}
