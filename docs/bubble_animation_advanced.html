d3 = require("d3@7")
Code 
width = Math.min(1000, window.innerWidth - 40)
height = 700
centerX = width * 0.5
centerY = height * 0.5

// Size scale
maxValue = d3.max(data, d => d.value)
radiusScale = d3.scaleSqrt()
  .domain([0, maxValue])
  .range([0, 60])
Code 
// Create nodes
nodes = data.map(d => ({
  ...d,
  x: centerX + (Math.random() - 0.5) * width * 0.8,
  y: centerY + (Math.random() - 0.5) * height * 0.8,
  r: 0,
  radius: radiusScale(d.value),
  fx: null,
  fy: null
}))
Code 
// Force simulation
simulation = {
  const sim = d3.forceSimulation(nodes)
    .force('charge', d3.forceManyBody().strength(5))
    .force('collide', d3.forceCollide(d => d.r + 3))
    .force('x', d3.forceX(centerX).strength(0.05))
    .force('y', d3.forceY(centerY).strength(0.05))
    .alphaDecay(0.015);
  
  return sim;
}

// Main chart with image patterns
chart = {
  const svg = d3.create("svg")
    .attr("width", width)
    .attr("height", height)
    .style("background", "#1a1a2e")
    .style("border-radius", "15px")
    .style("box-shadow", "0 10px 40px rgba(0,0,0,0.3)");

  // Create defs for patterns
  const defs = svg.append("defs");

  // Create pattern for each node
  nodes.forEach((d, i) => {
    const pattern = defs.append("pattern")
      .attr("id", `image-pattern-${i}`)
      .attr("width", 1)
      .attr("height", 1)
      .attr("patternContentUnits", "objectBoundingBox");

    pattern.append("image")
      .attr("xlink:href", d.imageUrl)
      .attr("width", 1)
      .attr("height", 1)
      .attr("preserveAspectRatio", "xMidYMid slice");
  });

  let focusedNode = null;

  // Create node groups
  const node = svg.selectAll(".node")
    .data(nodes)
    .join("g")
    .attr("class", "node")
    .style("cursor", "pointer")
    .call(d3.drag()
      .on("start", dragStarted)
      .on("drag", dragged)
      .on("end", dragEnded));

  // Add circles with image fill
  node.append("circle")
    .attr("id", d => `circle-${d.id}`)
    .attr("r", 0)
    .style("fill", (d, i) => `url(#image-pattern-${i})`)
    .style("stroke", "#4a90e2")
    .style("stroke-width", 3)
    .style("filter", "drop-shadow(0 4px 8px rgba(0,0,0,0.3))")
    .transition()
    .duration(2000)
    .delay((d, i) => i * 100)
    .ease(d3.easeElasticOut)
    .attr("r", d => d.radius);

  // Add name labels below bubbles
  node.append("text")
    .attr("class", "node-label")
    .attr("text-anchor", "middle")
    .attr("dy", d => d.radius + 20)
    .style("font-size", "12px")
    .style("font-weight", "600")
    .style("fill", "#fff")
    .style("opacity", 0.8)
    .style("pointer-events", "none")
    .text(d => d.name);

  // Add info overlay
  const overlay = node.append("foreignObject")
    .attr("class", "circle-overlay")
    .attr("x", d => -d.radius)
    .attr("y", d => -d.radius)
    .attr("width", d => d.radius * 2)
    .attr("height", d => d.radius * 2)
    .style("display", "none")
    .style("pointer-events", "none");

  overlay.append("xhtml:div")
    .style("width", "100%")
    .style("height", "100%")
    .style("display", "flex")
    .style("flex-direction", "column")
    .style("align-items", "center")
    .style("justify-content", "center")
    .style("background", "rgba(26, 26, 46, 0.95)")
    .style("border-radius", "50%")
    .style("color", "white")
    .style("padding", "20px")
    .style("text-align", "center")
    .style("border", "3px solid #4a90e2")
    .html(d => `
      <div style="font-size: 24px; font-weight: bold; margin-bottom: 8px; color: #4a90e2;">${d.name}</div>
      <div style="font-size: 14px; opacity: 0.7; margin-bottom: 10px; text-transform: uppercase; letter-spacing: 1px;">${d.category}</div>
      <div style="font-size: 13px; line-height: 1.5;">${d.desc}</div>
      <div style="font-size: 12px; margin-top: 10px; opacity: 0.6;">Proficiency: ${d.value}%</div>
    `);

  // Click handler
  node.on("click", function(event, d) {
    event.stopPropagation();
    
    if (d === focusedNode) return;

    // Reset previous
    if (focusedNode) {
      resetNode(focusedNode);
    }

    focusedNode = d;
    
    // Fade others
    node.style("opacity", n => n === d ? 1 : 0.3);
    node.selectAll(".circle-overlay").style("display", "none");
    
    // Animate to center
    simulation.alphaTarget(0.2).restart();
    
    d3.transition()
      .duration(2000)
      .ease(d3.easePolyOut)
      .tween("moveIn", () => {
        const ix = d3.interpolateNumber(d.x, centerX);
        const iy = d3.interpolateNumber(d.y, centerY);
        const ir = d3.interpolateNumber(d.r, Math.min(centerY * 0.7, 200));
        
        return t => {
          d.x = ix(t);
          d.y = iy(t);
          d.r = ir(t);
          simulation.force('collide', d3.forceCollide(n => n.r + 3));
        };
      })
      .on("end", () => {
        simulation.alphaTarget(0);
        d3.select(this).select(".circle-overlay").style("display", "block");
      });
  });

  // Click outside to reset
  svg.on("click", function(event) {
    if (focusedNode && event.target === this) {
      resetNode(focusedNode);
      focusedNode = null;
      
      node.style("opacity", 1);
      node.selectAll(".circle-overlay").style("display", "none");
    }
  });

  function resetNode(d) {
    simulation.alphaTarget(0.2).restart();
    
    d3.transition()
      .duration(2000)
      .ease(d3.easePolyOut)
      .tween("moveOut", () => {
        const ir = d3.interpolateNumber(d.r, d.radius);
        return t => {
          d.r = ir(t);
          simulation.force('collide', d3.forceCollide(n => n.r + 3));
        };
      })
      .on("end", () => {
        simulation.alphaTarget(0);
      });
  }

  function dragStarted(event, d) {
    if (!event.active) simulation.alphaTarget(0.2).restart();
    d.fx = d.x;
    d.fy = d.y;
  }

  function dragged(event, d) {
    d.fx = event.x;
    d.fy = event.y;
  }

  function dragEnded(event, d) {
    if (!event.active) simulation.alphaTarget(0);
    d.fx = null;
    d.fy = null;
  }

  // Update on tick
  simulation.on("tick", () => {
    node.attr("transform", d => `translate(${d.x},${d.y})`);
    
    node.select("circle")
      .attr("r", d => d.r || d.radius);
    
    node.select(".node-label")
      .attr("dy", d => (d.r || d.radius) + 20);
    
    node.select(".circle-overlay")
      .attr("x", d => -(d.r || d.radius))
      .attr("y", d => -(d.r || d.radius))
      .attr("width", d => (d.r || d.radius) * 2)
      .attr("height", d => (d.r || d.radius) * 2);
  });

  // Animate radius growth
  nodes.forEach((d, i) => {
    setTimeout(() => {
      d.r = d.radius;
      simulation.force('collide', d3.forceCollide(n => n.r + 3));
    }, i * 100);
  });

  return svg.node();
}

Features

✅ Image-based bubbles using SVG patterns (like yesenia.io)
✅ Elastic entry animation with staggered timing
✅ Physics simulation with collision detection
✅ Click to zoom with smooth transitions
✅ Drag interaction to reposition bubbles
✅ Info overlay with detailed descriptions
✅ Fade effect on non-focused bubbles
✅ Dark theme matching modern design

Customization Guide

Use Your Own Images

Replace imageUrl with: - Local images: "./images/mylogo.png" - External URLs: Any HTTPS image URL - Icon libraries: DevIcons, Font Awesome, etc.

Adjust Colors

.style("stroke", "#4a90e2")  // Border color
.style("background", "#1a1a2e")  // SVG background

Change Animation Speed

.duration(2000)  // Milliseconds (2000 = 2 seconds)

Modify Bubble Sizes

radiusScale = d3.scaleSqrt()
  .domain([0, maxValue])
  .range([0, 60])  // Min and max radius
