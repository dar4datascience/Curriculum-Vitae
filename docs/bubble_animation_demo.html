d3 = require("d3@7")
Code 
width = 900
height = 600
centerX = width * 0.5
centerY = height * 0.5

// Size scale
maxValue = d3.max(data, d => d.value)
radiusScale = d3.scaleSqrt()
  .domain([0, maxValue])
  .range([0, 50])
Code 
// Create nodes with physics properties
nodes = data.map(d => ({
  ...d,
  x: centerX + (Math.random() - 0.5) * width * 0.8,
  y: centerY + (Math.random() - 0.5) * height * 0.8,
  r: 0,
  radius: radiusScale(d.value),
  fx: null,
  fy: null
}))
Code 
// Force simulation
simulation = {
  const sim = d3.forceSimulation(nodes)
    .force('charge', d3.forceManyBody().strength(5))
    .force('collide', d3.forceCollide(d => d.r + 2))
    .force('x', d3.forceX(centerX).strength(0.05))
    .force('y', d3.forceY(centerY).strength(0.05))
    .alphaDecay(0.02);
  
  return sim;
}

// Main chart
chart = {
  const svg = d3.create("svg")
    .attr("width", width)
    .attr("height", height)
    .style("background", "linear-gradient(135deg, #667eea 0%, #764ba2 100%)")
    .style("border-radius", "10px");

  let focusedNode = null;

  // Create node groups
  const node = svg.selectAll(".node")
    .data(nodes)
    .join("g")
    .attr("class", "node")
    .style("cursor", "pointer")
    .call(d3.drag()
      .on("start", dragStarted)
      .on("drag", dragged)
      .on("end", dragEnded));

  // Add circles
  node.append("circle")
    .attr("id", d => `circle-${d.id}`)
    .attr("r", 0)
    .style("fill", "#fff")
    .style("stroke", "#333")
    .style("stroke-width", 2)
    .transition()
    .duration(2000)
    .ease(d3.easeElasticOut)
    .attr("r", d => d.radius)
    .on("end", function(d) {
      d3.select(this.parentNode).select(".node-icon").style("opacity", 1);
    });

  // Add emoji icons
  node.append("text")
    .attr("class", "node-icon")
    .attr("text-anchor", "middle")
    .attr("dy", "0.35em")
    .style("font-size", d => `${d.radius * 0.8}px`)
    .style("opacity", 0)
    .style("pointer-events", "none")
    .text(d => d.icon);

  // Add info overlay
  const overlay = node.append("foreignObject")
    .attr("class", "circle-overlay")
    .attr("x", d => -d.radius)
    .attr("y", d => -d.radius)
    .attr("width", d => d.radius * 2)
    .attr("height", d => d.radius * 2)
    .style("display", "none")
    .style("pointer-events", "none");

  overlay.append("xhtml:div")
    .style("width", "100%")
    .style("height", "100%")
    .style("display", "flex")
    .style("flex-direction", "column")
    .style("align-items", "center")
    .style("justify-content", "center")
    .style("background", "rgba(0, 0, 0, 0.85)")
    .style("border-radius", "50%")
    .style("color", "white")
    .style("padding", "10px")
    .style("text-align", "center")
    .html(d => `
      <div style="font-size: 18px; font-weight: bold; margin-bottom: 5px;">${d.name}</div>
      <div style="font-size: 12px; opacity: 0.8;">${d.category}</div>
      <div style="font-size: 11px; margin-top: 5px;">${d.desc}</div>
    `);

  // Click handler
  node.on("click", function(event, d) {
    event.stopPropagation();
    
    if (d === focusedNode) return;

    // Reset previous focused node
    if (focusedNode) {
      resetNode(focusedNode);
    }

    focusedNode = d;
    
    // Hide all overlays and fade other nodes
    node.selectAll(".circle-overlay").style("display", "none");
    node.selectAll(".node-icon").style("opacity", 0.3);
    
    // Animate to center and grow
    simulation.alphaTarget(0.2).restart();
    
    d3.select(this).select(".node-icon").style("opacity", 0.1);
    
    d3.transition()
      .duration(2000)
      .ease(d3.easePolyOut)
      .tween("moveIn", () => {
        const ix = d3.interpolateNumber(d.x, centerX);
        const iy = d3.interpolateNumber(d.y, centerY);
        const ir = d3.interpolateNumber(d.r, Math.min(centerY * 0.6, 150));
        
        return t => {
          d.x = ix(t);
          d.y = iy(t);
          d.r = ir(t);
          simulation.force('collide', d3.forceCollide(n => n.r + 2));
        };
      })
      .on("end", () => {
        simulation.alphaTarget(0);
        d3.select(this).select(".circle-overlay").style("display", "block");
      });
  });

  // Click outside to reset
  svg.on("click", function(event) {
    if (focusedNode && event.target === this) {
      resetNode(focusedNode);
      focusedNode = null;
      
      // Show all icons
      node.selectAll(".node-icon").style("opacity", 1);
      node.selectAll(".circle-overlay").style("display", "none");
    }
  });

  function resetNode(d) {
    simulation.alphaTarget(0.2).restart();
    
    d3.transition()
      .duration(2000)
      .ease(d3.easePolyOut)
      .tween("moveOut", () => {
        const ir = d3.interpolateNumber(d.r, d.radius);
        return t => {
          d.r = ir(t);
          simulation.force('collide', d3.forceCollide(n => n.r + 2));
        };
      })
      .on("end", () => {
        simulation.alphaTarget(0);
      });
  }

  function dragStarted(event, d) {
    if (!event.active) simulation.alphaTarget(0.2).restart();
    d.fx = d.x;
    d.fy = d.y;
  }

  function dragged(event, d) {
    d.fx = event.x;
    d.fy = event.y;
  }

  function dragEnded(event, d) {
    if (!event.active) simulation.alphaTarget(0);
    d.fx = null;
    d.fy = null;
  }

  // Update positions on tick
  simulation.on("tick", () => {
    node.attr("transform", d => `translate(${d.x},${d.y})`);
    
    node.select("circle")
      .attr("r", d => d.r || d.radius);
    
    node.select(".circle-overlay")
      .attr("x", d => -d.r)
      .attr("y", d => -d.r)
      .attr("width", d => d.r * 2)
      .attr("height", d => d.r * 2);
  });

  // Initial animation
  nodes.forEach((d, i) => {
    setTimeout(() => {
      d.r = d.radius;
      simulation.force('collide', d3.forceCollide(n => n.r + 2));
    }, i * 100);
  });

  return svg.node();
}

How It Works

This interactive visualization uses:

  • D3.js Force Simulation: Physics-based layout with collision detection
  • Observable JS: Reactive programming in Quarto
  • Smooth Transitions: Elastic and polynomial easing functions
  • Interactive Elements: Click to zoom, drag to move, click outside to reset

Key Features

  1. Entry Animation: Bubbles pop in with elastic bounce effect
  2. Physics Simulation: Continuous floating motion with collision avoidance
  3. Click Interaction: Zoom to center and display details
  4. Drag Support: Move bubbles around manually
  5. Reset: Click background to return to normal state

Customization
